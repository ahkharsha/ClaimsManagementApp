2. The Core Server Logic (Days 1 & 2)Create a server.js file. This handles both the WebSocket connections from Candidate 2 (React) and the incoming HTTP Webhooks from Candidate 1 & 3 (Spring Boot / Python).JavaScriptrequire('dotenv').config();
const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const cors = require('cors');

const app = express();
app.use(cors());
app.use(express.json());

const server = http.createServer(app);

// Initialize Socket.IO with CORS for the React frontend
const io = new Server(server, {
  cors: {
    origin: "*", // In production, restrict this to the S3 React URL
    methods: ["GET", "POST"]
  }
});

// --- SOCKET.IO CONNECTION LOGIC ---
io.on('connection', (socket) => {
  console.log(`New client connected: ${socket.id}`);

  // When React connects, it should emit 'register' with user details
  socket.on('register', ({ userId, role }) => {
    if (userId) {
      socket.join(userId); // Join personal room (for direct alerts)
      console.log(`Socket ${socket.id} joined personal room: ${userId}`);
    }
    if (role) {
      socket.join(role); // Join role room (for broadcast alerts like 'claims_manager')
      console.log(`Socket ${socket.id} joined role room: ${role}`);
    }
  });

  socket.on('disconnect', () => {
    console.log(`Client disconnected: ${socket.id}`);
  });
});

// --- REST API FOR BACKEND TO TRIGGER NOTIFICATIONS ---
// Spring Boot or Python will call this endpoint
app.post('/api/notify', (req, res) => {
  const { event, payload, targetUserId, targetRole } = req.body;

  if (!event || !payload) {
    return res.status(400).json({ error: 'Event and payload are required' });
  }

  // Route to a specific user (e.g., policy renewal for Customer A)
  if (targetUserId) {
    io.to(targetUserId).emit(event, payload);
    console.log(`Emitted '${event}' to User: ${targetUserId}`);
  } 
  // Route to a specific role (e.g., new claim filed for Claims Managers)
  else if (targetRole) {
    io.to(targetRole).emit(event, payload);
    console.log(`Emitted '${event}' to Role: ${targetRole}`);
  } 
  // Broadcast to everyone (fallback/system alerts)
  else {
    io.emit(event, payload);
    console.log(`Broadcasted '${event}' to all connected clients`);
  }

  res.status(200).json({ success: true, message: 'Notification dispatched' });
});

const PORT = process.env.PORT || 4000;
server.listen(PORT, () => {
  console.log(`Notification Service running on port ${PORT}`);
});
